\begin{theorem}[Hockey Stick Identity]
    For all $n,k\in\N_0$, we have:
    \begin{equation*}
        \binom{k}{k} + \binom{k+1}{k} + \binom{k+2}{k} + \cdots + \binom{n}{k} = \binom{n+1}{k+1}
    \end{equation*}
\end{theorem}

\begin{align*}
    \int_{-\sqrt{n}}^{\sqrt{n}} \qty(1-\frac{x^2}{n})^n \dx
    &= \sqrt{n}\int_{-1}^1 \qty(1-y^2)^n \dy
        \tag{$x=\sqrt{n}y$} \\
    &= \sqrt{n}\int_{-\frac{\pi}{2}}^{\frac{\pi}{2}} \cos(t)^{2n+1} \dt
        \tag{$y=\sin(t)$} \\
\end{align*}

\begin{align*}
    \int_{-\infty}^{\infty} \qty(1+\frac{x^2}{n})^{-n} \dx
    &= \sqrt{n}\int_{-\infty}^{\infty} \qty(1+y^2)^{-n} \dy
        \tag{$x=\sqrt{n}y$} \\
    &= \sqrt{n}\int_{-\frac{\pi}{2}}^{\frac{\pi}{2}} \cos(t)^{2n-2} \dt
        \tag{$y=\tan(t)$} \\
\end{align*}

\begin{equation*}
    \int_0^{\tfrac{\pi}{2}} \sin(x)^{2m} \dx = \frac{(2m)!}{2^{2m+1}(m!)^2}\pi
\end{equation*}
\begin{equation*}
    \int_0^{\tfrac{\pi}{2}} \sin(x)^{2m+1} \dx = \frac{2^{2m}(m!)^2}{(2m+1)!}
\end{equation*}
Comparing $2m$ with $2m+1$:
\begin{equation*}
    \frac{(2m)!}{2^{2m+1}(m!)^2}\pi > \frac{2^{2m}(m!)^2}{(2m+1)!}
\end{equation*}
\begin{equation*}
    \pi > \frac{2^{4m+1}(m!)^4}{(2m)!(2m+1)!}
\end{equation*}
Comparing $2m$ with $2m-1$:
\begin{equation*}
    \frac{(2m)!}{2^{2m+1}(m!)^2}\pi  < \frac{2^{2m-2}((m-1)!)^2}{(2m-1)!}
\end{equation*}
\begin{equation*}
    \pi < \frac{2^{4m-1}(m!)^2((m-1)!)^2}{(2m)!(2m-1)!}
\end{equation*}
\begin{equation*}
    \frac{}{(2m-1)!}\frac{(2m+1)!}{4m^2} > 1
\end{equation*}
