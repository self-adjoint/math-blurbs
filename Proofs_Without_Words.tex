\begin{tikzpicture}
    \pgfmathsetmacro{\width}{6}
    \pgfmathsetmacro{\height}{8}
    \pgfmathsetmacro{\x}{atan(3/4)}
    \pgfmathsetmacro{\xplusy}{atan(\height/(\width*sec(\x)^2-\height*tan(\x)))}
    \coordinate (O) at (0,0);
    \coordinate (A) at (\width,{\width*tan(\x)});
    \coordinate (B) at ({\width*sec(\x)^2-\height*tan(\x)},\height);
    \coordinate (W) at (\width,0);
    \coordinate (H) at (0,\height);
    \coordinate (Z) at (\width,\height);
    % \draw (O) rectangle (Z);
    % \draw (O) -- (A) -- (B) -- cycle;
    \draw[Maroon] (0:1) arc(0:\x:1) node[midway, anchor={180+\x/2}]{$x$};
    \draw[Blue] (\x:1) arc(\x:\xplusy:1) node[midway, anchor={180+(\x+\xplusy)/2}]{$y$};
    \draw[YellowOrange] ($(A)+(0,1)$) arc(90:{90+\x}:1) node[midway, anchor={-90+\x/2}]{$x$};
    \draw[ForestGreen] ($(B)+(-1,0)$) arc (180:{180+\xplusy}:1) node[midway, anchor={\xplusy/2}]{$x+y$};
    \draw ($(A)+({180+\x}:0.3)$) --++ ({90+\x}:0.3) --++ (\x:0.3);
    \draw (O) -- (B) node[midway, anchor={-\xplusy}, inner sep=8pt]{$1$};
    \draw[Blue] (O) -- (A) node[midway, sloped, above]{$\cos(y)$};
    \draw[Blue] (A) -- (B) node[midway, sloped, above]{$\sin(y)$};
    \draw[ForestGreen] (O) -- (H) node[midway, left]{$\sin(x+y)$};
    \draw[ForestGreen] (H) -- (B) node[midway, above]{$\cos(x+y)$};
    \draw[Maroon] (O) -- (W) node[midway, below]{$\cos(x)\cos(y)$};
    \draw[Maroon] (W) -- (A) node[midway, right]{$\sin(x)\cos(y)$};
    \draw[YellowOrange] (B) -- (Z) node[midway, above]{$\sin(x)\sin(y)$};
    \draw[YellowOrange] (A) -- (Z) node[midway, right]{$\cos(x)\sin(y)$};
\end{tikzpicture}

$\sin(x+y)=\sin(x)\cos(y)+\cos(x)\sin(y)$

$\cos(x+y)=\cos(x)\cos(y)-\sin(x)\sin(y)$
