\begin{theorem}[Hockey Stick Identity]
    For all $n,k\in\N_0$, we have:
    \begin{equation*}
        \binom{k}{k} + \binom{k+1}{k} + \binom{k+2}{k} + \cdots + \binom{n}{k} = \binom{n+1}{k+1}
    \end{equation*}
\end{theorem}
